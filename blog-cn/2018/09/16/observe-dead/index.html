监听应用程序被杀死

监听应用进入后台

[[NSNotificationCenter defaultCenter] addObserver:self selector:@selector(sh_comeHome:) name:@"UIApplicationDidEnterBackgroundNotification" object:nil];

(void)sh_comeHome:(UIApplication *)application {
        NSLog(@"进入后台");
 }

监听应用程序被杀死

[[NSNotificationCenter defaultCenter] addObserver:self selector:@selector(applicationWillTerminate:) name:@"UIApplicationWillTerminateNotification" object:nil];

(void)applicationWillTerminate:(UIApplication *)application {
       NSLog(@"程序被杀死");
}
